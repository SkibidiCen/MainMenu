local function filterButtons(query)
    local searchText = string.lower(query)
    local buttons = {}
    local positionIndex = 0

    for _, button in pairs(buttonContainer:GetChildren()) do
        if button:IsA("TextButton") then
            if searchText == "" then
                button.Visible = true
                button.Position = UDim2.new(0, 0, 0, positionIndex * 50)
                positionIndex = positionIndex + 1
            else
                local buttonText = string.lower(button.Text)
                if string.find(buttonText, searchText) then
                    table.insert(buttons, button)
                else
                    button.Visible = false
                end
            end
        end
    end

    -- Reorder buttons based on search query
    if searchText ~= "" then
        table.sort(buttons, function(a, b)
            return string.find(string.lower(a.Text), searchText) and not string.find(string.lower(b.Text), searchText)
        end)

        for i, button in ipairs(buttons) do
            button.Position = UDim2.new(0, 0, 0, (i - 1) * 50)
            button.Visible = true
        end
    end
end

-- Connect the filter function to the search box's text changed event
searchBox:GetPropertyChangedSignal("Text"):Connect(function()
    filterButtons(searchBox.Text)
end)

-- Function to change button text color and update labels
local function changeButtonTextColor(button, currentlyWorkingText)
    for _, btn in pairs(buttonContainer:GetChildren()) do
        if btn:IsA("TextButton") then
            if btn == button then
                local gradient = Instance.new("UIGradient")
                gradient.Color = ColorSequence.new{
                    ColorSequenceKeypoint.new(0, Color3.fromRGB(11, 232, 190)),
                    ColorSequenceKeypoint.new(1, Color3.fromRGB(74, 255, 225))
                }
                gradient.Parent = btn
                btn.TextColor3 = Color3.fromRGB(74, 255, 225)
            else
                btn.TextColor3 = Color3.fromRGB(255, 255, 255)
                local gradient = btn:FindFirstChildOfClass("UIGradient")
                if gradient then
                    gradient:Destroy()
                end
            end
        end
    end
    currentlyWorkingLabel.Text = currentlyWorkingText
end

-- Function to handle the "Run Script" button click
runButton.MouseButton1Click:Connect(function()
    if scriptToRun ~= "" then
        local success, errorMessage = pcall(function()
            local scriptContent = game:HttpGet(scriptToRun)
            if scriptContent then
                local func, loadError = loadstring(scriptContent)
                if func then
                    func()
                else
                    warn("Loadstring error: " .. loadError)
                end
            else
                warn("Failed to fetch script from URL: " .. scriptToRun)
            end
        end)
        if success then
            mainFrame:TweenSize(UDim2.new(0, 0, 0, 0), Enum.EasingDirection.In, Enum.EasingStyle.Quart, 0.5, true)
            wait(0.5)
            screenGui:Destroy()
        else
            warn("Error executing script: " .. errorMessage)
        end
    else
        print("No script selected")
    end
end)

-- Tools Frame: Add customizable buttons
local function createCustomButton(parent, name, position, color, textColor, textSize, scriptUrl)
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0, 140, 0, 40)
    button.Position = position
    button.BackgroundColor3 = color
    button.TextColor3 = textColor
    button.Text = name
    button.Font = Enum.Font.Gotham
    button.TextSize = textSize
    button.Parent = parent

    local uiGradient = Instance.new("UIGradient")
    uiGradient.Color = ColorSequence.new{
        ColorSequenceKeypoint.new(0, Color3.fromRGB(61, 121, 212)),
        ColorSequenceKeypoint.new(1, Color3.fromRGB(65, 148, 211))
    }
    uiGradient.Parent = button

    local isConfirming = false
    local originalText = name

    local function resetButton()
        button.Text = originalText
        isConfirming = false
    end

    button.MouseButton1Click:Connect(function()
        if isConfirming then
            resetButton()
            if scriptUrl ~= "" then
                local success, errorMessage = pcall(function()
                    loadstring(game:HttpGet(scriptUrl))()
                end)
                if not success then
                    warn("Error running script: " .. errorMessage)
                end
            else
                warn("No script provided!")
            end
        else
            button.Text = "Are You Sure?"
            button.TextScaled = true
            isConfirming = true
            wait(6)
            if isConfirming then
                resetButton()
            end
        end
    end)
end
